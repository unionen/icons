# Unionen Icon Pack

An icon pack for FormKit inputs. Based on [@formkit/icons](https://github.com/formkit/formkit).

## Installation

    pnpm add -D @unionen/icons

## Usage

    // formkit.config.ts

    import { icons } from '@unionen/icons';

    const config: DefaultConfigOptions = {
      icons: {
        ...icons
      },
    }
